依赖注入通过set方式注入实现【见Model：Spring-02-di】

什么是依赖？bean对象的创建依赖于容器

什么是注入？bean对象的所有属性都由容器注入

一个Student类的注入

```java
public class Student {

    String name;
    Address address;
    String[] books;
    List<String> hobbies;
    Map card;
    Set<String> games;
    String wife;
    Properties info;
}
```
对应的xml
``` xml
<bean id="address" class="com.acn.Address">
        <property name="address" value="芜湖"/>
</bean>

<bean id="student" class="com.acn.Student">

    <!--普通值-->
    <property name="name" value="acn"/>

    <!--Bean-->
    <property name="address" ref="address"/>

    <!--数组-->
    <property name="books">
        <array>
            <value>书1</value>
            <value>书2</value>
            <value>书3</value>
        </array>
    </property>

    <!--List-->
    <property name="hobbies">
        <list>
            <value>敲代码</value>
        </list>
    </property>

    <!--Map-->
    <property name="card">
        <map>
            <entry key="身份证" value="123"/>
            <entry key="学生证" value="456"/>
        </map>
    </property>

    <!--Set-->
    <property name="games">
        <set>
            <value>LOL</value>
            <value>COC</value>
            <value>BOB</value>
        </set>
    </property>

    <!--null-->
    <property name="wife">
        <null/>
    </property>

    <!--properties-->
    <property name="info">
        <props>
            <prop key="学号">123</prop>
            <prop key="班级">114514</prop>
        </props>
    </property>

</bean>
```

通过命名空间简化xml
``` xml
<!--p命名空间，不要注入非基本类型的属性-->
<!--这个p:card="身份证:123,学生证:456"就注入不了，
报错java.lang.IllegalStateException: Cannot convert value of type 'java.lang.String' to required type 'java.util.Map' for property 'card'
-->
<bean id="stu" class="com.acn.Student"
      p:name="aca"
      p:address-ref="address"
      p:books="书1,书2"
      p:hobbies="兴趣1,兴趣2"
      p:games="LOL,COC,BOB"
      p:info="学号:123,班级:114514"
>
</bean>


<bean id="addressC" class="com.acn.Address"
      c:address="构造器注入地址"
/>
```

ATTENTION：
- 使用命名空间需要导入xml约束
`xmlns:p="http://www.springframework.org/schema/p"`
- 使用p命名空间需要该属性提供Setter
- 使用c命名空间需要满足构造器的参数
